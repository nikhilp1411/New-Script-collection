## Please create this script in /bin/ folder to avoid path dependencies for execution.

#!/bin/bash

for f in *

do
	new="${f// /_}"
	if [ "$new" != "$f" ]
	then
		if [ -e "$new" ]
		then
		echo not renaming \""$f"\" because \""$new"\" 
already exists
	else
		echo moving "$f" to "$new"
	mv "$f" "$new"
	fi
fi
done

## "${file//[^a-zA-Z0-9_- ]/}" -- to remove special char and spaces in file name
