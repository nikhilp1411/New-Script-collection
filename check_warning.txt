#!/bin/bash

countfiles=$(ls /tmp/ | wc -l)

if (($countfiles<=10)); then
                echo "OK - $countfiles services are ok"
                exit 0
        elif ((11<=$countfiles && $countfiles<=20)); then
                echo "WARNING STATE - $countfiles"
                exit 1
        elif ((20<=$countfiles)); then
                echo "CRITICAL STATE - $countfiles"
                exit 2
        else
                echo "UNKNOWN - $countfiles"
                exit 3
fi
